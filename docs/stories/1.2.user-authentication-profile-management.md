interface User {
  id: string; // UUID primary key

  cookingSkillLevel: "beginner" | "intermediate" | "advanced";
  preferredMealComplexity: "simple" | "moderate" | "complex";

    };
  };
  rotationResetCount: number; // Recipe rotation tracking
  createdAt: Date;
  updatedAt: Date;
}
